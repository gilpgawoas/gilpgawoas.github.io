<?php

require_once __DIR__ . "/../lib/php/ejecuta.php";
require_once __DIR__ . "/../lib/php/leeTexto.php";
require_once __DIR__ . "/../lib/php/validaToken.php";

ejecuta(function () {

 session_start();

 $token = leeTexto("token");
 validaToken("formulario", $token);

 // Si el token se hall√≥, precesa normalmente la forma.
 $saludo = leeTexto("saludo");
 $nombre = leeTexto("nombre");
 if ($saludo === "") {
  throw new Exception("Falta el saludo");
 }
 if ($nombre === "") {
  throw new Exception("Falta el nombre");
 }
 $resultado =
  "{$saludo} {$nombre}.";
 return $resultado;
});
