</head>

<body>
 <form id="forma" action="javascript:envia()">
  <h1>Mensajes</h1>
  <p>
   <label>
    Alias
    <input name="alias">
   </label>
  </p>
  <p>
   <label>
    Mensaje
    <input name="mensaje">
   </label>
  </p>
  <p>
   <button>Enviar</button>
  </p>
 </form>
 <pre id="pre"></pre>
 <script>
  try {
   /* Cambia la url por la de tu
    * servicio. */
   const ws = new WebSocket(
    "wss://phpmensajes.gilbertopacheco.repl.co")
   ws.onopen = wsAbierto
   ws.onmessage = recibeMensaje
   ws.onerror = error
   ws.onclose = cerrado

   /** @param {MessageEvent} evt */
   function recibeMensaje(evt) {
    pre.textContent +=
     evt.data + "\n\n"
   }

   function envia() {
    const msg =
     `${forma.alias.value}
${forma.mensaje.value}`
    ws.send(msg)
   }

   function wsAbierto() {
    console.log('Conectado')
   }

   function error(evt) {
    console.error(evt)
    alert("Error en la conexión.")
   }

   function cerrado(evt) {
    console.log(evt)
    alert("Conexión terminada.")
   }
  } catch (e) {
   console.log(e)
   alert(e.message)
  }
 </script>
