4require_once __DIR__ . "/../lib/php/ejecutaServicio.php";

11require_once __DIR__ . "/CUE.php";
12require_once __DIR__ . "/ROL_IDS.php";
13require_once __DIR__ . "/Bd.php";
14require_once __DIR__ . "/protege.php";
15
16ejecutaServicio(function () {
17
18 $sesion = protege();
19
20 if ($sesion->cue !== "")
21 throw new ProblemDetails(
22 status: NO_AUTORIZADO,
23 type: "/error/sesioniniciada.html",
24 title: "Sesión iniciada.",
25 detail: "La sesión ya está iniciada.",
26 );
27
28 $cue = recuperaTexto("cue");
29 $match = recuperaTexto("match");
30
31 $cue = validaCue($cue);
32
33 if ($match === false)
34 throw new ProblemDetails(
35 status: BAD_REQUEST,
36 title: "Falta el match.",
37 type: "/error/faltamatch.html",
38 detail: "La solicitud no tiene el valor de match.",
39 );
40
41 if ($match === "")
42 throw new ProblemDetails(
43 status: BAD_REQUEST,
44 title: "Match en blanco.",
45 type: "/error/matchenblanco.html",
46 detail: "Pon texto en el campo match.",
47 );
48
49 $pdo = Bd::pdo();
50
51 $usuario =
52 selectFirst(pdo: $pdo, from: USUARIO, where: [USU_CUE => $cue]);
53
54 if ($usuario === false || !password_verify($match, $usuario[USU_MATCH]))
55 throw new ProblemDetails(
56 status: BAD_REQUEST,
57 type: "/error/datosincorrectos.html",
58 title: "Datos incorrectos.",
59 detail: "El cue y/o el match proporcionados son incorrectos.",
60 );
61
62 $rolIds = fetchAll(
63 $pdo->query(
64 "SELECT ROL_ID
65 FROM USU_ROL
66 WHERE USU_ID = :USU_ID
67 ORDER BY USU_ID"
68 ),
69 [":USU_ID" => $usuario[USU_ID]],
70 PDO::FETCH_COLUMN
71 );
72
73 $_SESSION[CUE] = $cue;
74 $_SESSION[ROL_IDS] = $rolIds;
75 devuelveJson([
76 CUE => $cue,
77 ROL_IDS => $rolIds
78 ]);
79});
