  import {
   muestraError
  } from "./lib/js/muestraError.js"
  import {
   MetodoHttp
  } from "./lib/js/MetodoHttp.js"
  import {
   enviaJson
  } from "./lib/js/enviaJson.js"

  btnEnviaJson.addEventListener(
   "click", enviaJsonPost)

  async function enviaJsonPost() {
   try {
    const datos = {
     saludo: "Hola",
     nombre: "pp"
    }
    const resultado =
     await enviaJson(datos,
      MetodoHttp.POST,
      "srv/SrvJson.php")
    alert(resultado)
   } catch (e) {
    muestraError(e)
   }
  }
