 <h1>Servicios</h1>

  Env√≠a JSON
 </button>
 <form id="forma">
  <p>
   <label>
    Nombre:
    <input name="nombre">
   </label>
  </p>
  <p>
   <button>Procesa</button>
  </p>
 </form>

 <script type="module">
  import {
   ejecuta
  } from "./lib/js/ejecuta.js"
  import {
   muestraError
  } from "./lib/js/muestraError.js"
  import {
   submit
  } from "./lib/js/submit.js"
  import {
   Method
  } from "./lib/js/Method.js"
  import {
   enviaJson
  } from "./lib/js/enviaJson.js"

  btnError.addEventListener(
   "click", error)
  btnResultado.addEventListener(
   "click", resultado)
  btnResultado.addEventListener(
   "click", resultado)
  btnEnviaJson.addEventListener(
   "click", enviaJsonPost)
  forma.addEventListener(
   "submit", procesa)

  async function error() {
   try {
    await ejecuta(
     fetch("srv/SrvError.php"))
    alert("Ejecutado.")
   } catch (e) {
    muestraError(e)
   }
  }

  async function resultado() {
   try {
    const json = await ejecuta(
     fetch("srv/SrvResultado.php"))
    alert(`Nombre: ${json.nombre}
Mensaje: ${json.mensaje}`)
   } catch (e) {
    muestraError(e)
   }
  }

  async function enviaJsonPost() {
   try {
    const datos = {
     saludo: "Hola",
     nombre: "pp"
    }
    const json =
     await enviaJson(datos,
      Method.POST,
      "srv/SrvJson.php")
    alert(json.resultado)
   } catch (e) {
    muestraError(e)
   }
  }

  /** @param {Event} evt */
  async function procesa(evt) {
   try {
    const json = await submit(evt,
     "srv/SrvProcesa.php", forma)
    alert(json.resultado)
   } catch (e) {
    muestraError(e)
   }
  }
 </script>
