5class OrmPdo
6{
7 private string $nombreDeTabla;
8 /**
9 * @var string[]
10 */
11 private array $nombresDeLlaves;
12 /**
13 * @var string[]
14 */
15 private array $nombresDeCampos;
16 private bool $autoincrement;
17 private string $sequence;
18
19 /**
20 * @param string[] $nombresDeLlaves
21 * @param string[] $nombresDeCampos
22 */
23 public function __construct(
24 string $nombreDeTabla,
25 array $nombresDeLlaves,
26 array $nombresDeCampos,
27 bool $autoincrement = true,
28 string $sequence = "",
29 ) {
30 $this->nombreDeTabla = $nombreDeTabla;
31 $this->nombresDeLlaves = $nombresDeLlaves;
32 $this->autoincrement = $autoincrement;
33 $this->sequence = $sequence;
34 $this->nombresDeCampos = $nombresDeCampos;

36
37 /**
38 * @param array[] $objetos
39 */
40 public function insert(PDO $conexion, array $objetos)
41 {
42
43 $sqlCampos = implode(",", $this->nombresDeCampos);
44 $sqlValoresCampos =
45 implode(",", $this->convierteCamposEnParametros($this->nombresDeCampos));
46
47 if ($this->autoincrement) {
48
49 $sql =
50 "INSERT INTO $this->nombreDeTabla ($sqlCampos) VALUES ($sqlValoresCampos)";
51
52 /**
53 * @var string[]
54 */
55 $llaves = [];
56 $statement = $conexion->prepare($sql);
57 foreach ($objetos as $objeto) {
58 $this->ejecutaStatement($statement, $objeto);
59 if ($this->sequence === "") {
60 $id = $conexion->lastInsertId();
61 } else {
62 $id = $conexion->lastInsertId($this->sequence);
63 }
64 if ($id === false) {
65 $llaves[] = "";
66 } else {
67 $llaves[] = $id;
68 }
69 }
70 return $llaves;
71 } else {
72
73 $sqlLlaves = implode(",", $this->nombresDeLlaves);
74 $sqlValoresLlaves =
75 implode(",", $this->convierteCamposEnParametros($this->nombresDeLlaves));
76 $sql =
77 "INSERT INTO $this->nombreDeTabla ($sqlLlaves,$sqlCampos)
78 VALUES ($sqlValoresLlaves,$sqlValoresCampos)";
79
80 $statement = $conexion->prepare($sql);
81 foreach ($objetos as $objeto) {
82 $this->ejecutaStatement($statement, $objeto);
83 }
84 return [];
85 }
86 }
87
88 /**
89 * @param array[] $objetos
90 */
91 public function update(PDO $conexion, array $objetos)
92 {
93
94 $sqlCampos =
95 implode(",", $this->convierteCamposEnAsignaciones($this->nombresDeCampos));
96 $sqlLlaves = implode(
97 " AND ",
98 $this->convierteCamposEnAsignaciones($this->nombresDeLlaves)
99 );
100 $sql = "UPDATE $this->nombreDeTabla SET $sqlCampos WHERE $sqlLlaves";
101
102 $statement = $conexion->prepare($sql);
103 foreach ($objetos as $objeto) {
104 $this->ejecutaStatement($statement, $objeto);
105 }
106 }
107
108 /**
109 * @param array[] $laves
110 */
111 public function delete(PDO $conexion, array $llaves)
112 {
113
114 $sqlLlaves = implode(
115 " AND ",
116 $this->convierteCamposEnAsignaciones($this->nombresDeLlaves)
117 );
118 $sql = "DELETE FROM $this->nombreDeTabla WHERE $sqlLlaves";
119
120 $statement = $conexion->prepare($sql);
121 foreach ($llaves as $llave) {
122 $this->ejecutaStatement($statement, $llave);
123 }
124 }
125
126 /**
127 * @param array $valor
128 */
129 public function select(
130 PDO $conexion,
131 array $lave,
132 int $mode = PDO::FETCH_ASSOC,
133 $opcional = null
134 ) {
135
136 $sqlLlaves = implode(
137 " AND ",
138 $this->convierteCamposEnAsignaciones($this->nombresDeLlaves)
139 );
140 $sql = "SELECT * FROM $this->nombreDeTabla WHERE $sqlLlaves";
141
142 $statement = $conexion->prepare($sql);
143 $parametros = $this->convierteAsignacionEnParametros($lave);
144 return fetch($statement, $parametros, $mode, $opcional);
145 }
146
147 /**
148 * @param string[] $campos
149 */
150 private function convierteCamposEnParametros(array $campos)
151 {
152 $parametros = [];
153 foreach ($campos as $campo) {
154 $parametros[] = ":$campo";
155 }
156 return $parametros;
157 }
158
159 /**
160 * @param string[] $campos
161 */
162 private function convierteCamposEnAsignaciones(array $campos)
163 {
164 $parametros = [];
165 foreach ($campos as $campo) {
166 $parametros[] = "$campo=:$campo";
167 }
168 return $parametros;
169 }
170
171 private function convierteAsignacionEnParametros(array $asignacion)
172 {
173 $parametros = [];
174 foreach ($asignacion as $llave => $valor) {
175 $parametros[":$llave"] = $valor;
176 }
177 return $parametros;
178 }
179
180 private function ejecutaStatement(PDOStatement $statement, array $asignacion)
181 {
182 $statement->execute($this->convierteAsignacionEnParametros($asignacion));
183 }
184}
