3/**
4 * @param { Document | HTMLElement } raizHtml
5 * @param {HTMLInputElement} input
6 */
7export function muestraImagenSeleccionada(raizHtml, input) {
8 return new Promise((resolve, reject) => {
9 setTimeout(() => {
10
11 const img = getImgParaElementoHtml(raizHtml, input)
12 if (img !== null) {
13 try {
14
15 const dataUrl = getDataUrlDeSeleccion(input)
16 if (dataUrl === "") {
17
18 const file = input.dataset.file
19 if (file === undefined || file === "") {
20 img.hidden = true
21 img.src = ""
22 } else {
23 img.hidden = false
24 img.src = file
25 }
26
27 } else {
28
29 img.hidden = false
30 img.src = dataUrl
31
32 }
33
34 resolve(true)
35
36 } catch (error) {
37
38 img.hidden = true
39
40 reject(error)
41
42 }
43 }
44
45 },
46 500)
47 })
48}

52/**
53 * @param {HTMLInputElement} input
54 */
55export function getDataUrlDeSeleccion(input) {
56 const seleccion = getArchivoSeleccionado(input)
57 if (seleccion === null) {
58 return ""
59 } else {
60 return URL.createObjectURL(seleccion)
61 }
62}
63// Permite que los eventos de html usen la función.
64window["getDataUrlDeSeleccion"] = getDataUrlDeSeleccion

67/**
68 * @param { HTMLInputElement } input
69 */
70export function getArchivoSeleccionado(input) {
71 const seleccion = input.files
72 if (seleccion === null || seleccion.length === 0) {
73 return null
74 } else {
75 return seleccion.item(0)
76 }
77}
