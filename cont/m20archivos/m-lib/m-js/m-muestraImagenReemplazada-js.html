import { getDataUrlDeSeleccion } from "./getDataUrlDeSeleccion.js"

/**
 * @param {HTMLInputElement} input
 * @param {HTMLImageElement} img
 */
export async function muestraImagenReemplazada(input, img) {
 try {
  const dataUrl = await getDataUrlDeSeleccion(input)
  if (dataUrl === '') {
   const imagenInicial = img.dataset.inicial
   if (imagenInicial === undefined || imagenInicial === '') {
    img.hidden = true
   } else {
    img.hidden = false
    img.src = imagenInicial
   }
  } else {
   img.hidden = false
   img.src = dataUrl
  }
 } catch (error) {
  img.hidden = true
  throw error
 }
}

// Permite que los eventos de html usen la funci√≥n.
window["muestraImagenReemplazada"] = muestraImagenReemplazada
