<?php
class Pasatiempo
{
 public $id;
 public $nombre;
 public function valida(
  string $msgFaltaElNombre
 ) {
  if (!$this->nombre) {
   throw new Exception(
    $msgFaltaElNombre
   );
  }
 }
 public function validaCambio(
  string $msgFaltaElId,
  string $msgFaltaElNombre
 ) {
  if (!$this->id) {
   throw new Exception(
    $msgFaltaElId
   );
  }
  if (!$this->nombre) {
   throw new Exception(
    $msgFaltaElNombre
   );
  }
 }
}
