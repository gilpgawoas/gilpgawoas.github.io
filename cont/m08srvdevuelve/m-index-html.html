 <p><button onclick="resultado()">Resultado</button></p>

 <script type="module">

  import { ejecuta } from "./lib/js/ejecuta.js"
  import { muestraError } from "./lib/js/muestraError.js"

  async function resultado() {
   try {
    const json = await ejecuta(
     "srv/srvResultado.php")
    alert(`Nombre: ${json.nombre}
Mensaje: ${json.mensaje}`)
   } catch (error) {
    muestraError(error)
   }
  }
  // Permite que los eventos de html usen la función.
  window["resultado"] = resultado

 </script>
