<?php

require_once __DIR__ . "/../modelo/Amigo.php";
require_once __DIR__ . "/AccesoBd.php";

function amigoAgrega(Amigo $modelo)
{
 $modelo->valida();
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "INSERT INTO AMIGO
    (AMI_NOMBRE, PAS_ID)
   VALUES
    (:nombre, :pasId)"
 );
 $stmt->execute([
  ":nombre" => $modelo->nombre,
  ":pasId" => $modelo->pasatiempo === null
   ? null
   : $modelo->pasatiempo->id
 ]);
 /* Si usas una secuencia para generar el id,
  * pasa como par√°metro de lastInsertId el
  * nombre de dicha secuencia. */
  $modelo->id = $con->lastInsertId();
 }
