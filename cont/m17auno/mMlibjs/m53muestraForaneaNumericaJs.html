import {
 muestraNumero
} from "./muestraNumero.js"

/**
 * @template { {value:string} } E
 * @param {E} elementoHtml
 * @param {any} modelo
 * @param {string} campoId
 */
export function
 muestraForaneaNumerica(
  elementoHtml, modelo, campoId) {
 if (modelo === undefined
  || modelo === null) {
  elementoHtml.value = ""
 } else {
  const id = modelo[campoId]
  muestraNumero(elementoHtml, id)
 }
}
