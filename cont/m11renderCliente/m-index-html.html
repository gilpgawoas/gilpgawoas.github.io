22 <script type="module">
23
24 import { consumeJson } from "./lib/js/consumeJson.js"
25 import { htmlentities } from "./lib/js/htmlentities.js"
26 import { muestraError } from "./lib/js/muestraError.js"
27
28 async function descargaDatos() {
29 try {
30 const respuesta = await consumeJson("srv/lista.php")
31 // Genera el código HTML de la lista.
32 let render = ""
33 for (const modelo of respuesta.body) {
34 /* Codifica nombre y color para que cambie los caracteres especiales y
35 * el texto no se pueda interpretar como HTML. Esta técnica evita la
36 * inyección de código. */
37 const nombre =
38 typeof modelo.nombre === "string" ? htmlentities(modelo.nombre) : ""
39 const color =
40 typeof modelo.color === "string" ? htmlentities(modelo.color) : ""
41 render += /* html */
42 `<dt>${nombre}</dt>
43 <dd>${color}</dd>`
44 }
45 lista.innerHTML = render
46 } catch (error) {
47 muestraError(error)
48 }
49 }
50 // Permite que los eventos de html usen la función.
51 window["descargaDatos"] = descargaDatos
52
53 </script>
54
55</body>
