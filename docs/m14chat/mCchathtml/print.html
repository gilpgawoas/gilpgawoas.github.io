2. index.html

<!DOCTYPE html>
<html>

<head>

 <meta charset="UTF-8">
 <meta name="viewport" content="width=device-width">

 <title>Chat</title>

 <script type="module">

  import { muestraError } from "./lib/js/muestraError.js"
  import { errorEnWebSocket } from "./lib/js/errorEnWebSocket.js"
  import { webSocketCerrado } from "./lib/js/webSocketCerrado.js"

  try {

   // Cambia la url por la de tu servicio. */
   const ws = new WebSocket("wss://chatwss.gilbertopachec5.repl.co/")

   // Se invoca cuando se ha establecido una conexión al servidor.
   ws.onopen = () => {
    pre.textContent = "Conectado.\n"
    console.log("Conectado")
   }

   // Se invoca al presentarse un error en la conexión al servidor.
   ws.onerror = errorEnWebSocket

   // Se invoca cuando se cierra la conexión al servidor.
   ws.onclose = webSocketCerrado

   // Se invoca al recibir un mensaje del servidor.
   ws.onmessage = event => pre.textContent += event.data + "\n\n"

   // Permite que los eventos de html usen la variable.
   window["ws"] = ws

  } catch (error) {
   muestraError(error)
  }

 </script>

</head>

<body>

 <form id="forma" onsubmit="
   try {
     event.preventDefault()
     const mensaje = `${inputAlias.value.trim()}
${inputMensaje.value.trim()}`
     ws.send(mensaje)
   } catch (error) {
    muestraError(error)
   }
  ">

  <h1>Chat</h1>

  <p>
   <label>

 </form>

</body>

</html>

/** @param {Event} evt */
export function errorEnWebSocket(evt) {
 console.error(evt)
 alert("Error en la conexión.")
}

// Permite que los eventos de html usen la función.
window["errorEnWebSocket"] = errorEnWebSocket

2. lib / js / muestraError.js

/**
 * Muestra un error en la consola y en un cuadro de
 * alerta el mensaje de una excepción.
 * @param { Error | null } error descripción del error.
 */
export function muestraError(error) {
 if (error === null) {
  console.log("Error")
  alert("Error")
 } else {
  console.error(error)
  alert(error.message)
 }
}

// Permite que los eventos de html usen la función.
window["muestraError"] = muestraError
