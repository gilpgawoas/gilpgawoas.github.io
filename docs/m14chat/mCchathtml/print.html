1. index.html

<!DOCTYPE html>
<html>

<head>
 <meta charset="UTF-8">
 <meta name="viewport"
   content="width=device-width">
 <title>Chat</title>
</head>

<body>
 <form id="forma">
  <h1>Chat</h1>
  <p>
   <label>
    Alias
    <input id="alias" required>
   </label>
  </p>
  <p>
   <label>
    Mensaje
    <input id="mensaje" required>
   </label>
  </p>
  <p>
   <button>Enviar</button>
  </p>
 </form>
 <pre id="pre"></pre>

 <script>
  try {
   /* Cambia la url por la de tu
    * servicio. */
   const ws = new WebSocket(
    "wss://chatwss.gilbertopachec5.repl.co/")
   ws.onopen = abre
   ws.onmessage = recibeMensaje
   ws.onerror = error
   ws.onclose = cierra

   /** @param {MessageEvent} evt */
   function recibeMensaje(evt) {
    pre.textContent +=
     evt.data + "\n\n"
   }

   function abre() {
    console.log('Conectado')
    forma.addEventListener(
     "submit", envia)
   }

   function error(evt) {
    console.error(evt)
    alert("Error en la conexión.")
   }

   function cierra(evt) {
    console.log(evt)
    alert("Conexión terminada.")
    forma.removeEventListener(
     "submit", envia)
   }

   /** @param {Event} evt */
   function envia(evt) {
    try {
     evt.preventDefault()
     const msg =
      `${alias.value.trim()}
${mensaje.value.trim()}`
     ws.send(msg)

    } catch (e) {
     muestraError(e)
    }
   }
  } catch (e) {
   muestraError(e)
  }

  /** @param {Error} e */
  function muestraError(e) {
   console.log(e)
   alert(e.message)
  }
 </script>
</body>

</html>
